// This program loads 200 into the AC register. It then
// adds -1 to the AC register 200 times until the AC 
// register is 0. Then it writes the value 3 to
// memory location 230.
//
load_const -1
move_to_mbr
load_const 200
add
ifgo 3
load_const 230
move_to_mar
load_const 3
move_to_mbr
write_at_addr
exit
